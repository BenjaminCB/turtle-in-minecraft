block = WOOD
length = 20
width = 10
hight = 6

BuildHouse(block, length, width, hight)
BuildHouse(WOOD, 20, 10, 6)

//Build 4 identical houses side by side
repeat 4 do
	TurtleStart = position()
	StartDirection = faceing()
	BuildHouse(block, length, width, hight)
	goto(TurtleStart)
	turn(StartDirection)
	turn(right)
	forward(width)  //width+NUM to make spaceing in between
	turn(left)
end

// Builds house of the given dimentions, walkable space will be -2 in every direction
// ^^The total space used is described by the given dimentions.
function BuildHouse(block, length, width, hight) do
	ACTIVE_BLOCK = block
	TurtleStart = position()		//type: Position
	StartDirection = faceing()		//type: AbsoluteDirection

	Floor(block, length, width)

	up()
	repeat 2 do
		Wall(block, length, hight)
		faceing = StartDirection
		forward(length-1)
		turn(right)

		Wall(block, width, hight)
		faceing = StartDirection
		turn(right)
		forward(width-1)
		turn(right)
	end

	goto(TurtleStart)
	placing = false
	up(hight)
	placing = true

	Floor(block, length, width)

	goto(TurtleStart)
end

//function Floor(block, length, width, hight) to make it more consistent
function Floor(block, length, width)
	StartPosition = position()
	StartDirection = faceing()
	while length > 0 do
		forward(length-1)
		turn(right)
		forward(width)
		goto(StartPosition)
		turn(left)
		length -= 1
	end

end


function Wall(block, length, hight) do
	StartingPoint = position()
	repeat hight do
		forward(length-1)
		up()
		turn(right)
		turn(right)
	end
	goto(StartingPoint)
end
